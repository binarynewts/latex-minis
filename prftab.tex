\newcounter{currprfstep}
\newcounter{case}
\newcounter{currcasestep}

\newcommand*{\prfstart}{\begin{tabbing}\setcounter{currprfstep}{1}\setcounter{case}{0}}
\newcommand{\prfend}{\end{tabbing}}
\newcommand*{\prfstep}{
    \ifnum\thecurrprfstep=1
        \\~~\=(1)~~\=
    \else
        \\\> (\thecurrprfstep)\>
    \fi
    \stepcounter{currprfstep}
}

\newcommand*{\casestart}{\\[3ex]\stepcounter{case}\setcounter{currcasestep}{1}\textbf{Case \Roman{case}:}~}
\newcommand*{\casestep}{
    \ifnum\thecurrcasestep=1
        \\\>\> (\Roman{case}.\thecurrcasestep)~~\=
    \else
        \\\>\> (\Roman{case}.\thecurrcasestep)\>
    \fi
    \stepcounter{currcasestep}
}

\newcommand*{\caserefer}[1]{(\Roman{case}.\the\numexpr \value{currcasestep} - #1 \relax)}
\newcommand*{\prfrefer}[1]{(\the\numexpr \value{currprfstep} - #1 \relax)}
